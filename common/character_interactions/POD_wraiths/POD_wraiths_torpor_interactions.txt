POD_remove_caul_interaction = {
	category = interaction_category_wraith
	icon = POD_wraith_interaction
	desc = POD_remove_caul_interaction_desc
	
	cost = { piety = 50 }

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:recipient = { POD_is_wraith_trigger = yes }
		scope:recipient = { has_character_modifier = caul_modifier }
		scope:recipient = { has_trait = torpor }
		POD_basic_interaction_is_shown = yes
	}
	
	is_valid_showing_failures_only = {
		scope:actor = {
			trigger_if = {
				limit = {
					NOT = { has_trait = wraith }
					NOT = { has_trait = demon }
				}
				has_access_to_underworld_trigger = yes
			}
		}
		POD_interaction_is_valid_showing_failures_only = yes
	}
	
	on_accept = {
		scope:recipient = {
			remove_character_modifier = caul_modifier
			remove_trait = torpor
            if = {
                limit = {
					scope:actor = { has_trait = wraith }
				}
                set_character_faith = scope:actor.faith
            }
		}
	}
	
	auto_accept = yes
	
	ai_will_do = {
		base = 0
	}
}
